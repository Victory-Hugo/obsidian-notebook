导入 NEX 文件

为什么 mtDNA 需要导入 NEX 而 Y 染色体可以直接导入对齐的 FASTA？

因为 nex 文件可以帮助更快速、方便的分区。

制作 NEX 文件

准备文件：已经对齐了的 FASTA 文件。

代码

以下 python 根据预定的规则进行生成分区 nex 文件：


# 定义输入和输出文件路径
input_file_path = "最终版本ALN无hap extraction.fasta"
output_file_path = "ALN.nex"

def read_fasta(file_path):
    sequences = {}
    with open(file_path, 'r') as file:
        sequence_id = ''
        sequence_data = ''
        for line in file:
            line = line.strip()
            if line.startswith('>'):
                if sequence_id:
                    sequences[sequence_id] = sequence_data
                sequence_id = line[1:]
                sequence_data = ''
            else:
                sequence_data += line
        if sequence_id:
            sequences[sequence_id] = sequence_data
    return sequences

def write_nexus(sequences, file_path):
    num_taxa = len(sequences)
    num_characters = len(next(iter(sequences.values())))
    
    with open(file_path, 'w') as file:
        file.write("#NEXUS\n\n")
        file.write("Begin data;\n")
        file.write(f"Dimensions ntax={num_taxa} nchar={num_characters};\n")
        file.write("Format datatype=dna missing=? gap=-;\n")
        file.write("Matrix\n\n")
        
        for sequence_id, sequence_data in sequences.items():
            file.write(f"{sequence_id}\t{sequence_data}\n")
        
        file.write(";\n")
        file.write("End;\n\n")
        
        file.write("begin assumptions;\n")
        file.write("charset DLOOP = 1-599 16001-16594;\n")
        file.write("charset CODING = 600-16000;\n")
        file.write("end;\n")

# 读取FASTA文件并写入NEXUS格式文件
sequences = read_fasta(input_file_path)
write_nexus(sequences, output_file_path)

print(f"文件已成功转换并保存为: {output_file_path}")
