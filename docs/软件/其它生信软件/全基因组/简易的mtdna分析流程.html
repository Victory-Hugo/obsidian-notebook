分步操作

为参考序列 fasta 建立字典和索引

# 建立字典
gatk CreateSequenceDictionary R=/home/luolintao/10K_HGDP/BAM2VCF/rcrs.fasta O= /home/luolintao/10K_HGDP/BAM2VCF/rcrs.dict
# 建立索引
samtools faidx rcrs.fasta

从 BAM 文件中提取 mtDNA

# 建立索引
samtools index 原始的文件.bam
# 提取
samtools view -b 原始的文件.bam chrM > chrM.bam
samtools index chrM.bam

产生 VCF 文件

gatk --java-options "-Xmx4G" HaplotypeCaller \
-R rcrs.fasta \
-I chrM.bam \
-O output.vcf \
-L chrM \
-ploidy 2

一步脚本

#!/bin/bash

# 定义输入和输出目录
input_dir="/home/luolintao/10K_HGDP"
output_dir="/home/luolintao/10K_HGDP/BAM2VCF"

# 创建输出目录
mkdir -p $output_dir

# 加载所需的模块或定义工具的路径
# module load samtools
# module load gatk

# 遍历所有的BAM文件
for bam_file in $input_dir/*.bam; do
    echo "Processing $bam_file"
    
    # 1. 建立索引
    samtools index $bam_file

    # 2. 提取mtDNA并建立索引
    base_name=$(basename $bam_file .bam)
    samtools view -b $bam_file chrM > $output_dir/${base_name}_chrM.bam
    samtools index $output_dir/${base_name}_chrM.bam

    # 3. 产生VCF文件
    gatk --java-options "-Xmx4G" HaplotypeCaller \
    -R rcrs.fasta \
    -I $output_dir/${base_name}_chrM.bam \
    -O $output_dir/${base_name}.vcf \
    -L chrM \
    -ploidy 2
    echo "Preceessing OK"
done

echo "All processing complete!"

单倍群分型产生 fasta 文件(手动)

在进行这一步之前，请一定要进入 Haplogrep 的安装文件夹。cd 你的安装路径。

./haplogrep3 classify \
--input /mnt/c/Users/victo/Desktop/FINAL_VERSION_DEL33.fasta \
--tree phylotree-rcrs@17.2 \
--output /mnt/c/Users/victo/Desktop/分型结果.txt \
--write-fasta # 这个选项不是必须，如果你选了，那么就会生成一个全长的fasta文件。如果你是全测，那么可以勾选，否则会产生虚假的fasta序列。
注意这个软件的--tree 选项只能在以下挑选！
phylotree-rcrs@17.2
phylotree-rcrs@17.1
phylotree-rcrs@17.0
phylotree-rcrs@16.0
phylotree-rcrs@15.0
phylotree-rsrs@17.1
phylotree-fu-rcrs@1.2
phylotree-fu-rcrs@1.1
phylotree-fu-rcrs@1.0

单倍群分型产生 fasta 文件(脚本)

#!/bin/bash

# 设置输入和输出目录
input_dir="输入文件的路径"
output_dir="输出文件的路径"

# 检查输出目录是否存在，如果不存在则创建
mkdir -p "$output_dir"
cd 进入你的软件安装路径
# 遍历目录中的所有 VCF 文件
for vcf_file in "$input_dir"/*.vcf; do
    # 获取不带路径的文件名
    filename=$(basename "$vcf_file")
    # 构建输出文件名
    output_file="${output_dir}/${filename%.vcf}_分型结果.txt"

    # 执行 classify 命令
    ./haplogrep3 classify \
        --input "$vcf_file" \
        --tree phylotree-rcrs@17.0 \
        --output "$output_file" \
        --write-fasta

    # 输出处理的文件名
    echo "Processed: $vcf_file"
done

echo "All files processed."
